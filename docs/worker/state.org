
**** Worker State

This file is rather short as it just stores the states to be shared
around throughout the web worker.

#+begin_src clojurescript :tangle ../../src/worker/state.cljs
(ns worker.state)

(defonce !client (atom nil))
(defonce !media-cache (atom {}))
(defonce !plugin-handlers (atom {}))
(defn register-handler [id handler-fn]
  (swap! !plugin-handlers assoc id handler-fn))

#+end_src
