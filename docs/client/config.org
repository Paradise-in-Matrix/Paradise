(defn load-config []
  (-> (net/fetch "./config.edn"  #js {:cache "no-store"} )
      (p/then #(.text %))
      (p/then #(read-string %))))

#+begin_src clojurescript :tangle ../../src/client/config.cljs
(defn load-i18n []
  (-> (net/fetch "./i18n.edn" #js {:cache "no-store"})
      (p/then #(.text %))
      (p/then (fn [raw]
                (let [dict (cljs.reader/read-string raw)]
                  (re-frame/dispatch-sync [:i18n/set-dictionary dict])
                  dict)))))

#+end_src


This is to ensure that we never cache this fetch... so we
can always update the version if needed.
#+begin_src clojurescript :tangle ../../src/client/config.cljs
(defn check-remote-version []
  (let [url (str "./config.edn?t=" (.getTime (js/Date.)))]
    (-> (net/fetch url #js {:cache "no-store"})
        (p/then #(.text %))
        (p/then #(read-string %))
        (p/then :version)
        (p/catch (fn [_] nil)))))
#+end_src
